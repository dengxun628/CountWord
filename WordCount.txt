#include <stdio.h>
#include <stdlib.h>
int CountWord(char str1[])
{
	int j = 0,countW = 0;

	while (str1[j] != '\0')
	{
		
		if (str1[j] != ' ' && str1[j] != ',')
		{
			j++;
			continue;
		}
		else
		{
			countW++;
			j++;
		}
	}
	return ++countW;
}
int CountCh(char str[])
{
	int sum = 0;
	int i = 0;
	while (str[i] != '\0')
	{
		sum++;
		i++;
	}
	return sum;
}
